<static> Marc.Converters.subclassImportConverter(recordOrType, overrides)
Returns a constructor derived from the appropriate import converter. This allows client translators to subclass converters. 
var converterClass = Marc.Converters.subclassImportConverter(myrecord, {
		//Include some non-standard subfields
		_getAbstract : function(record, item) {
			var value = record.getValue(
				[Marc.Unimarc.Tags.DISSERTATION_THESIS_NOTE,
		         Marc.Unimarc.Tags.SUMMARY_OR_ABSTRACT], 
				"abced",
	    		{fieldJoin: "\n"});
			if(value) {
				item.abstractNote = value;
			};
		},
		//Call superclass
		_getTitle : function(record, item) {
			Marc.UnimarcImportConverter.prototype.call(this, record, item);
			item.title = "[ADDED BY SUBCLASS]" + item.title;
		}
});
var converter = new converterClass();
var item = converter.convert(myrecord);
Parameters:
{String|Marc.Record} recordOrType
the Marc type ("unimarc" or "marc21") or a record to be converted. In the latter case, the Marc type will be auto-detected from the record's fields.
{Object} overrides Optional
an object containing method overrides
